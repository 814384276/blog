###### 4.构造函数

```javascript
function Person(name, age) {
	this.name = name;
    this.age = age;
    this.say = function() {
        
    }
}

let obj = new Person('xiaoming', 18)
let obj2 = new Person('xiaohong', 12);

obj == obj2  // false (两个不同的实例对象，内存分别存)
obj.say == obj2.say   //false (内存浪费)

//a.当new时候做了什么：
/* 
	1.创建一个新的空对象
	2.将构造函数中的this指向新的空对象
	3.执行构造函数
	4.返回新的空对象
*/

//b.使用构造函数创建对象的缺点：
/*
	1.每一次通过new创建一个实例对象的时候，say方法每次都开辟一个内存(浪费内存)
*/


//解决say方法内存不共用问题
let objectMethod = {
    say: function say() {
    
	}
}

function Person(name, age) {
    this.name = name;
    this.age = age;
    this.say = objectMethod.say;
}

let obj = new Person();
let obj2 = new Person()

obj == obj2  //false
obj.say == obj2.say  //true (公用一个内存)



```
